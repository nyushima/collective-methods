# my silly regular expression

from bs4 import BeautifulSoup
import urllib2
import re

print('This script helps you to generate a Buenos Aires version of any song. Enter a URL of a song on azlyrics.com')
print('Example URL: http://www.azlyrics.com/lyrics/aha/takeonme.html')


url = raw_input("Enter an azlyrics.com URL")

if url == "":
    url = "http://www.azlyrics.com/lyrics/aha/takeonme.html"

print("The URL you entered is " + url)

opener = urllib2.build_opener()
opener.addheaders = [('User-agent', 'Mozilla/5.0 (Windows NT 6.1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/41.0.2228.0 Safari/537.36')] 
# i'm now adding a user-agent header to everything just in case any website refuse to work
response = opener.open(url)

page = response.read()
soup = BeautifulSoup(page, "html.parser")

def no_class_no_id(tag):
    return not tag.has_attr('class') and not tag.has_attr('id')

articlebody = soup.find_all("div", {"class": None})

for entity in articlebody:
    entity_text = entity.get_text()
    entity_text = re.sub('(Y|y)ou', 'dulce de leche', entity_text)
    entity_text = re.sub('(m|M)e', 'bife de lomo', entity_text)
    entity_text = re.sub('(T|t)hing', 'tango', entity_text)
    print (entity_text)
